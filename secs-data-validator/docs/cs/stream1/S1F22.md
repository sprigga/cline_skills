# S1F22 - Data Variable Namelist Reply

**Direction:** Sent by Equipment Only

---

| **S1F22** | Data Variable Namelist Reply | Sent by Equipment Only |

Comment: A:0 for DVVALNAME and UNITS indicates unknown VID or that VID is not a DVVAL. (proposed ballot item 4824B, Oct 2011)

Format:

{L:n
{L:3

[VID](http://hume.com/secs/items.html#VID)

[DVVALNAME](http://hume.com/secs/items.html#DVVALNAME)

[UNITS](http://hume.com/secs/items.html#UNITS)

 }}

```
    // S1F22 C# Parse reply
    SecsHost sp=(SecsHost)your_SecsHost;
    RcResult rcr = sp.SendSecsMsg(1, 21, true, sendData, true);
    string TSN_data;
    bool ok=true;
    while (ok) {   // break out of loop on error
        if (rcr.rc != 0) {ok=false; break;}
        TSN_data = rcr.result;
        // expect L:n {L:3 VID DVVALNAME UNITS}
        // variables for data items and parsing
        string VID;  	// A:n (varies)  A variable ID
        string DVVALNAME;  	// A:n (always)  a descriptive name for a Data Value variable (DVVAL)
        string UNITS;  	// A:n (always)  units identifier (see E5 Section 9)
        string [] list0;
        list0 = SecsPort.ListSplit(TSN_data);
        if (list0.Length < 1) { ok=false; break; }
        int n;
        for(n=1; n < list0.Length; n++) {
            string [] list1;
            list1 = SecsPort.ListSplit(list0[n]);
            if (list1.Length != 4 || list1[0] != "L:3") { ok=false; break; }
            string [] list2;
            list2 = SecsPort.ListSplit(list1[1]);
            if (list2.Length < 1) { ok = false; break; }
            if (list2.Length == 2) { VID = list2[1];} else { VID = ""; }
            list2 = SecsPort.ListSplit(list1[2]);
            if (list2.Length < 1) { ok = false; break; }
            if (!list2[0].StartsWith("A:")) { ok = false; break; }
            if (list2.Length == 2) { DVVALNAME = list2[1];} else { DVVALNAME = ""; }
            list2 = SecsPort.ListSplit(list1[3]);
            if (list2.Length < 1) { ok = false; break; }
            if (!list2[0].StartsWith("A:")) { ok = false; break; }
            if (list2.Length == 2) { UNITS = list2[1];} else { UNITS = ""; }
            }
        if (!ok) break;
        return;  // parsed ok
        } // end while(ok)
```
