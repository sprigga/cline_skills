{L:3
[MID](http://hume.com/secs/items.html#MID)

[PPID](http://hume.com/secs/items.html#PPID)

[PPBODY](http://hume.com/secs/items.html#PPBODY) }

        ok = False

    if len(pylist1) < 1:
        ok = False
        break
    #(typePPBODY, lengthPPBODY) = pylist1[0].split(':')
    PPBODY = []
    for i in range(0, len(PPBODY)):

    return # finished ok
