            break

        if len(pylist2) < 1:
            ok = False
            break

            ok = False
            break
        for a in range(1, len(pylist2)):
            pylist3 = lsplit(pylist2[a])
            if len(pylist3) < 1:
                ok = False
                break
            #(typeMID, lengthMID) = pylist3[0].split(':')
            MID = (pylist3[1] if len(pylist3) == 2 else None)
        if not ok: break
    if not ok: break
