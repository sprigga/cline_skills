# S2F54 - Return Report Definitions

**Direction:** Equipment Only

---

| **S2F54** | Return Report Definitions | Equipment Only |

Comment: a is 0 for unknown RPTID, n is 0 for no reports defined and L:0 request, new 8/2021

Format:

{L:n
{L:2

[RPTID](http://hume.com/secs/items.html#RPTID)

{L:a

[VID](http://hume.com/secs/items.html#VID)

}

 }}

```
    // S2F54 C# Parse reply
    SecsHost sp=(SecsHost)your_SecsHost;
    RcResult rcr = sp.SendSecsMsg(2, 53, true, sendData, true);
    string TSN_data;
    bool ok=true;
    while (ok) {   // break out of loop on error
        if (rcr.rc != 0) {ok=false; break;}
        TSN_data = rcr.result;
        // expect L:n {L:2 RPTID {L:a VID}}
        // variables for data items and parsing
        string RPTID;  	// U4:1 (varies)  report ID
        string VID;  	// A:n (varies)  A variable ID
        string [] list0;
        list0 = SecsPort.ListSplit(TSN_data);
        if (list0.Length < 1) { ok=false; break; }
        int n;
        for(n=1; n < list0.Length; n++) {
            string [] list1;
            list1 = SecsPort.ListSplit(list0[n]);
            if (list1.Length != 3 || list1[0] != "L:2") { ok=false; break; }
            string [] list2;
            list2 = SecsPort.ListSplit(list1[1]);
            if (list2.Length < 1) { ok = false; break; }
            if (list2.Length == 2) { RPTID = list2[1];} else { RPTID = ""; }
            list2 = SecsPort.ListSplit(list1[2]);
            if (list2.Length < 1) { ok=false; break; }
            int a;
            for(a=1; a < list2.Length; a++) {
                string [] list3;
                list3 = SecsPort.ListSplit(list2[a]);
                if (list3.Length < 1) { ok = false; break; }
                if (list3.Length == 2) { VID = list3[1];} else { VID = ""; }
                }
            if (!ok) break;
            }
        if (!ok) break;
        return;  // parsed ok
        } // end while(ok)
```
